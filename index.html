CREATE TABLE IF NOT EXISTS public.meddelanden ( id uuid PRIMARY KEY DEFAULT gen_random_uuid(), avsandare_id uuid NOT NULL REFERENCES auth.users(id), mottagare_id uuid NOT NULL REFERENCES auth.users(id), text TEXT NOT NULL, lank_typ TEXT, lank_id uuid, lank_namn TEXT, last BOOLEAN DEFAULT false, created_at timestamptz DEFAULT now() ); ALTER TABLE public.meddelanden ENABLE ROW LEVEL SECURITY; -- Bara avsändare och mottagare ser meddelandet CREATE POLICY "se egna meddelanden" ON public.meddelanden FOR SELECT TO authenticated USING (auth.uid() = avsandare_id OR auth.uid() = mottagare_id); -- Bara skicka som sig själv CREATE POLICY "skicka meddelanden" ON public.meddelanden FOR INSERT TO authenticated WITH CHECK (auth.uid() = avsandare_id); -- Mottagare får uppdatera (markera läst) CREATE POLICY "uppdatera mottagna" ON public.meddelanden FOR UPDATE TO authenticated USING (auth.uid() = mottagare_id OR auth.uid() = avsandare_id);
